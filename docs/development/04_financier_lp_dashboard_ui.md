## Overview

Build the complete Liquidity Provider (Financier) dashboard for Seed Finance. LPs use this interface to:
- **Deposit USDC** to the liquidity pool and receive SEED shares
- **Monitor portfolio** value, share price, and yield performance
- **View analytics** on pool utilization, treasury allocation, and historical returns
- **Withdraw** their position (partially or fully)
- **Track transactions** and receive real-time notifications

### MVP Scope Decisions
- **Single pool** â€” No risk tranches (Junior/Mezz/Senior) for MVP
- **Crypto wallet** â€” RainbowKit/WalletConnect integration only (no Circle Wallet)
- **No KYC** â€” Skip accredited investor verification for testnet/MVP
- **Full dashboard** â€” Include yield analytics, transaction history, notifications

---

## Design System: Institution-Grade UI

> **Design Philosophy:** Clean, professional, and trustworthy. The interface should feel like a Bloomberg terminal meets modern fintech â€” data-dense but never cluttered, precise but approachable. Every pixel should convey competence and security.

// ============ Pool State ============
interface PoolState {
  // Core metrics
  totalAssets: bigint;              // Total USDC under management
  totalSupply: bigint;              // Total SEED shares
  sharePrice: bigint;               // Assets per share (scaled)

  // Allocation
  availableLiquidity: bigint;       // Liquid USDC in contract
  totalDeployed: bigint;            // USDC in active invoices
  totalInTreasury: bigint;          // USDC in treasury strategies

  // Rates (basis points)
  utilizationRate: number;          // deployed / totalAssets
  treasuryAllocationRate: number;   // treasury / totalAssets

  // Yield tracking
  totalInvoiceYield: bigint;        // Cumulative invoice yield
  totalTreasuryYield: bigint;       // Cumulative treasury yield

  // Pool info
  activeInvoices: number;
  liquidityBuffer: bigint;
  maxTreasuryAllocation: number;    // basis points

  // Metadata
  lastUpdated: number;              // timestamp
}

// ============ User Position ============
interface UserPosition {
  address: string;                  // Wallet address
  sharesOwned: bigint;              // SEED balance
  currentValue: bigint;             // Shares * share price

  // Cost basis (from deposits/withdrawals)
  totalDeposited: bigint;
  totalWithdrawn: bigint;
  netDeposits: bigint;

  // Yield
  unrealizedGain: bigint;           // currentValue - netDeposits
  unrealizedGainPercent: number;

  // Pool share
  poolOwnership: number;            // percentage of total supply

  // Proportional allocation
  proportionalDeployed: bigint;
  proportionalTreasury: bigint;
  proportionalLiquid: bigint;
}

// ============ Transaction ============
type TransactionType =
  | 'deposit'
  | 'withdrawal'
  | 'yield_invoice'
  | 'yield_treasury'
  | 'pool_event';

interface Transaction {
  id: string;
  type: TransactionType;
  hash: string;                     // Tx hash
  blockNumber: number;
  timestamp: number;

  // Amounts
  assetsAmount: bigint;             // USDC
  sharesAmount: bigint;             // SEED
  sharePrice: bigint;

  // Metadata
  invoiceId?: number;               // For yield events
  description: string;

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const period = searchParams.get('period') || '30d';
  const address = searchParams.get('address');

  // Get historical yield data from indexer
  const yieldHistory = await getYieldHistory(period);
  const currentAPY = calculateAPY(yieldHistory);

  // User-specific if address provided
  const userYield = address
    ? await getUserYield(address, period)
    : null;

  return Response.json({
    currentAPY,
    periodYield: sumYield(yieldHistory),
    invoiceYield: yieldHistory.invoice,
    treasuryYield: yieldHistory.treasury,
    history: yieldHistory.timeSeries,
    userYield,
  });
}

// app/api/liquidity/transactions/route.ts

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const address = searchParams.get('address');
  const type = searchParams.get('type');
  const limit = parseInt(searchParams.get('limit') || '20');
  const offset = parseInt(searchParams.get('offset') || '0');

  // Query indexed events
  const transactions = await getTransactions({
    address,
    type,
    limit,
    offset,
  });

  const total = await getTransactionCount({ address, type });

  return Response.json({
    transactions,
    pagination: {
      total,
      limit,
      offset,
      hasMore: offset + transactions.length < total,
    },
  });
}

## Security Considerations

- **Input Validation**: Sanitize all user inputs, validate amounts
- **Approval Limits**: Use exact approval amounts, not unlimited
- **Slippage Protection**: Warn users if share price changed during tx
- **Rate Limiting**: Rate limit API endpoints
- **XSS Prevention**: Sanitize any user-provided content
- **HTTPS Only**: Enforce HTTPS in production

---

## Performance Considerations

- **Code Splitting**: Split dashboard pages for faster initial load
- **Data Caching**: Use TanStack Query caching, stale-while-revalidate
- **Optimistic Updates**: Show immediate UI feedback on transactions
- **Lazy Loading**: Lazy load charts and heavy components
- **Image Optimization**: Use Next.js Image component
- **Bundle Analysis**: Monitor bundle size, tree-shake unused code
