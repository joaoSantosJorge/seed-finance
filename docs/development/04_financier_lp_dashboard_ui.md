## Overview

Build the complete Liquidity Provider (Financier) dashboard for Seed Finance. LPs use this interface to:
- **Deposit USDC** to the liquidity pool and receive sfUSDC shares
- **Monitor portfolio** value, share price, and yield performance
- **View analytics** on pool utilization, treasury allocation, and historical returns
- **Withdraw** their position (partially or fully)
- **Track transactions** and receive real-time notifications

### MVP Scope Decisions
- **Single pool** â€” No risk tranches (Junior/Mezz/Senior) for MVP
- **Crypto wallet** â€” RainbowKit/WalletConnect integration only (no Circle Wallet)
- **No KYC** â€” Skip accredited investor verification for testnet/MVP
- **Full dashboard** â€” Include yield analytics, transaction history, notifications

// app/api/liquidity/yield/route.ts

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const period = searchParams.get('period') || '30d';
  const address = searchParams.get('address');

  // Get historical yield data from indexer
  const yieldHistory = await getYieldHistory(period);
  const currentAPY = calculateAPY(yieldHistory);

  // User-specific if address provided
  const userYield = address
    ? await getUserYield(address, period)
    : null;

  return Response.json({
    currentAPY,
    periodYield: sumYield(yieldHistory),
    invoiceYield: yieldHistory.invoice,
    treasuryYield: yieldHistory.treasury,
    history: yieldHistory.timeSeries,
    userYield,
  });
}

// app/api/liquidity/transactions/route.ts

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const address = searchParams.get('address');
  const type = searchParams.get('type');
  const limit = parseInt(searchParams.get('limit') || '20');
  const offset = parseInt(searchParams.get('offset') || '0');

  // Query indexed events
  const transactions = await getTransactions({
    address,
    type,
    limit,
    offset,
  });

  const total = await getTransactionCount({ address, type });

  return Response.json({
    transactions,
    pagination: {
      total,
      limit,
      offset,
      hasMore: offset + transactions.length < total,
    },
  });
}

## Security Considerations

- **Input Validation**: Sanitize all user inputs, validate amounts
- **Approval Limits**: Use exact approval amounts, not unlimited
- **Slippage Protection**: Warn users if share price changed during tx
- **Rate Limiting**: Rate limit API endpoints
- **XSS Prevention**: Sanitize any user-provided content
- **HTTPS Only**: Enforce HTTPS in production
